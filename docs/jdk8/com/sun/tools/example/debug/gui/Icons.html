    static private byte[] wordsToBytes(int[] wordArray) {







        byte[] bytes = new byte[wordArray.length * 4];







        int inx = bytes.length;







        for (int i = wordArray.length-1; i >= 0; --i) {







            int word = wordArray[i];







            for (int j = 0; j < 4; ++j) {







                bytes[--inx] = (byte)(word & 0xff);







                word = word >>> 8;







            }







        }







        return bytes;







    }
